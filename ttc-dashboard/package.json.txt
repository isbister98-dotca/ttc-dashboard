// This file runs on Vercel's backend network
export default async function handler(request, response) {
    // 1. Get the command from the frontend (e.g., 'vehicleLocations' or 'routeConfig')
    const { command, r, s, t } = request.query;

    // 2. Build the official TTC URL
    let baseUrl = `https://webservices.umoiq.com/service/publicXMLFeed?a=ttc&command=${command}`;
    
    // Add optional parameters if they exist
    if (r) baseUrl += `&r=${r}`;
    if (s) baseUrl += `&s=${s}`;
    if (t) baseUrl += `&t=${t}`;

    try {
        // 3. Fetch data from TTC using Vercel's high-speed connection
        const serverRes = await fetch(baseUrl);
        const data = await serverRes.text();

        // 4. Send data back to your website with "Cache" headers
        // This makes it faster by telling the browser "this data is good for 5 seconds"
        response.setHeader('Cache-Control', 's-maxage=5, stale-while-revalidate');
        response.setHeader('Content-Type', 'text/xml');
        response.status(200).send(data);

    } catch (error) {
        response.status(500).json({ error: 'Failed to fetch TTC data' });
    }
}
