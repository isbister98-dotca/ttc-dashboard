// This file runs on Vercel's backend network
export default async function handler(request, response) {
    const { command, r, s, t } = request.query;

    // Build the official TTC data URL
    let baseUrl = `https://webservices.umoiq.com/service/publicXMLFeed?a=ttc&command=${command || 'vehicleLocations'}`;
    
    // Add optional filters (Route, Stop, or Time) if the dashboard asks for them
    if (r) baseUrl += `&r=${r}`;
    if (s) baseUrl += `&s=${s}`;
    if (t) baseUrl += `&t=${t}`;

    try {
        // Fetch the data from the TTC servers
        const serverRes = await fetch(baseUrl);
        const data = await serverRes.text();

        // Send the XML data back to your dashboard
        response.setHeader('Cache-Control', 's-maxage=5, stale-while-revalidate');
        response.setHeader('Content-Type', 'text/xml');
        response.status(200).send(data);

    } catch (error) {
        // If the TTC server is down, send an error message
        response.status(500).json({ error: 'Failed to fetch TTC data' });
    }
}
